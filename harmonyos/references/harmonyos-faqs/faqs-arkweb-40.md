Web组件的[onOverScroll](../harmonyos-references/arkts-basic-components-web-events.md#onoverscroll10)事件可以判断滑动是否到达顶部或底部。当yOffset小于0时，表示页面到达顶端；当yOffset大于0时，表示页面到达底端。

具体可参考代码示例：传递滑动事件可参考[Web组件嵌套滚动](../harmonyos-guides/web-nested-scrolling.md)，将Web组件嵌入可滚动容器中，使用[nestedScroll](../harmonyos-references/arkts-basic-components-web-attributes.md#nestedscroll11)属性设置嵌套滚动选项，实现滑动事件传递。具体代码示例如下：

```
1. import { webview } from '@kit.ArkWeb';

3. @Entry
4. @Component
5. struct Index {
6. private scrollerForScroll: Scroller = new Scroller();
7. controller: webview.WebviewController = new webview.WebviewController; // Web component controller, used for loading pages and interaction control
8. @State scrollDirection: ScrollDirection = ScrollDirection.Vertical;
9. title = '2023落下帷幕，一起迎接2024年***********************************************';
10. review = '评论';
11. context = '一起迎接2024年，***********************************************************************************';
12. list = [0, 0, 0, 0];

14. build() {
15. Flex() {
16. Scroll(this.scrollerForScroll) {
17. Column({ space: 5 }) {
18. Text(this.title)
19. .fontSize(40)
20. .padding(40)
21. List() {
22. ForEach(this.list, () => {
23. ListItem() {
24. Text(this.context).fontSize(16)
25. }
26. })
27. }

29. Web({ src: $rawfile('index.html'), controller: this.controller })
30. .nestedScroll({
31. scrollForward: NestedScrollMode.SELF_FIRST,
32. scrollBackward: NestedScrollMode.SELF_FIRST
33. })
34. .height('100%')
35. .width('100%')
36. .layoutMode(WebLayoutMode.FIT_CONTENT)
37. Text(this.review)
38. .fontSize(30)
39. List() {
40. ForEach(this.list, () => {
41. ListItem() {
42. Text(this.context)
43. .fontSize(16)
44. }
45. })
46. }
47. }
48. .width('95%')
49. }
50. .scrollBar(BarState.Off)
51. .width('100%')
52. .height('120%')
53. .scrollable(this.scrollDirection)
54. }
55. .width('100%')
56. .height('100%')

58. }
59. }
```
